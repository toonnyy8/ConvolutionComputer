for(let n = 0; n < (input_X.length + input_H.length - 1); n++)
{
    output_Y[n] = 0;

    {
        if(input_X[k] != undefined && input_H[n - k] != undefined)
        {
            output_Y[n] += input_X[k] * input_H[n - k];
        };
    };
};
